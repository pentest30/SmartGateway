Quick Start

# Start SQL Server
docker run -e "ACCEPT_EULA=Y" -e "MSSQL_SA_PASSWORD=YourStrong@Passw0rd" \
  -p 1433:1433 -d mcr.microsoft.com/mssql/server:2022-latest

# Start the gateway
cd src/SmartGateway.Host && dotnet run --urls http://localhost:5000

# Start the admin API
cd src/SmartGateway.Api && dotnet run --no-launch-profile --urls http://localhost:5002

# Create a route and test
curl -X POST http://localhost:5002/admin/api/clusters \
  -H "Content-Type: application/json" \
  -d '{"clusterId":"my-api","loadBalancing":"RoundRobin"}'

curl -X POST http://localhost:5002/admin/api/services/register \
  -H "Content-Type: application/json" \
  -d '{"clusterId":"my-api","destinationId":"inst-1","address":"http://localhost:3000"}'

curl -X POST http://localhost:5000/_admin/reload
curl http://localhost:5000/api/hello
